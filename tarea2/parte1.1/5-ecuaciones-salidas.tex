\usubsubsection{Ecuaciones de las salidas}
  \begin{align*}
    \tag{Salida $y_1$}
    y_1 = f(x_1, x_2, x_3, u_1) \\
    \tag{Salida $y_2$}
    y_2 = f(x_1, x_2, x_3, u_1)
  \end{align*}

  \noindent\textbf{Para $y_1$:}

    De \ref{eq:1.1:3:Nodovx}:
    \begin{align*}
      \tag{Salida $y_1$}
      \alignedbox{y_1}{= R_2 x_2 - R_2 x_3}
    \end{align*}

  \noindent\textbf{Para $y_2$:}

    Se sabe que $y_2$ es $x_2$.
    \begin{align*}
      \tag{Salida $y_2$}
      \alignedbox{y_2}{= x_2}
    \end{align*}
